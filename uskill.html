// Weighted Cosine Similarity v1.2
function compatScore(skill, src, tgt):
  S = W = 0; gaps = []

  for prim in skill.primitives:
    fvS  = featureVec(prim.id)
    fvT  = domainVec(tgt)
    sim  = cosine(fvS, fvT)
    w    = prim.critWeight
    S   += sim * w; W += w
    if sim < threshold: gaps.push(prim)

  base = BCM[src][tgt]
  return blend(S/W, base, 0.6, 0.4)
